//=============================================================================
// INPUTFILE FOR TOOL texmake
//
// offset into the BINDATA block are calculated using labels which are
// embedded in the BINDATA block.
//
//-----------------------------------------------------------------------------
// Keywords of texmaker:
//-----------------------------------------------------------------------------
//
// TEXCONFIG            ... Start of a TEXCONFIG block (followed by an optional
//                          name)
// STRIPn               ... Definition of the STRIP "n" in a TEXCONFIG
// NORMAL,MIRROR,IGNORE ... Valid values for mirrored field
// GENERICDATA          ... Start of a GENERICDATA block (followed by an optional
//                          name)
// BINDATA              ... Start of binary data. Values have to be separated by
//                          commas. Hexvalues have to start with "0x" and MUST
//                          consist of 2,4 or 8 hexdigits. Binary values have
//                          to start with 0b and MUST consist of 8, 16, or 32
//                          binary digits (0/1). You can also embedded strings
//                          in the BINDATA. Strings must start with a
//                          doublequote (") must end with a doublequote (").
//                          To embed a double quote within a string use (\").
//                          Other treated combinations are "\n", "\t", "\\",
//                          "\"" and "\xhh" where hh are hexdigits. At the end
//                          of a String a 0-byte is automatically appended.
// LABEL xyz            ... definition of label xyz
// NULL                 ... special value for offset (NULL offset)
//
//-----------------------------------------------------------------------------
// format of a STRIP definition:
//-----------------------------------------------------------------------------
//
// STRIPn offset_to_lightdata, offset_to_darkdata, offset_to_maskdata, mirrored
//
// Notes:
//
// (1) offsets can consist of any number of labels and integers separated by
//     '+' and '-' signs
// (2) 0 refers to the start of the BINDATA block. Use keyword NULL if you want
//     to set an offset to an REAL zero value (not the start of the BINDATA
//     block)
// (3) the parser will not accept offsets smaller than 0
//
//-----------------------------------------------------------------------------
// format of GENERICDATA definition (used to export any data block):
//-----------------------------------------------------------------------------
// GENERICDATA  [optional name]
// offset, length
//
//
//-----------------------------------------------------------------------------
// Optional Names of TEXCONFIG setups and GENERICDATA setups:
//-----------------------------------------------------------------------------
// keyword TEXCONFIG and GENERICDATA can be followed by an optional name. These
// names can be exported into a C headerfile by using "-i headerfilename" as
// additionally commandline parameters.
// Using optional names is suggested, because you haven't to use directly
// index numbers in your program. The generate headerfile will contain defines
// for all names and their related indices.
//
// Here just a small example how the headerfile may look like:
//
// #define TEXIDX_TICTLOGO 0
// #define TEXIDX_HOUSE1   1
// #define TEXIDX_HOUSE2   2
//
// .... and so on
//
//
//-----------------------------------------------------------------------------
// content of this file
//-----------------------------------------------------------------------------
//
// TEXCONFIG setups:
//
// index 0     .... monochrome tictlogo sprite
// index 1-2   .... house textures
// index 3-10  .... doomguy textures (8 directional sprite)
// index 11-12 .... textures for demo 3 (wall and door)
// index 13-15 .... textures for demo 6 (black texture,"demo by" and tictlogo)
// index 16-21 .... textures for spinning TICT-cube
//
//
// GENERICDATA blocks:
//
// index 0 ... demo 2 - background (4000 Bytes)
// index 1 ... demo 2 - hud mask (1152 Bytes)
// index 2 ... demo 3 - gui (4000 Bytes)
// index 3 ... demo 3 - hud mask (1152 Bytes)
// index 4 ... demo 3 - background  (2304 Bytes)
// index 5 ... demo 6 - screen 1 (4000 Bytes)
// index 6 ... demo 6 - screen 2 (4000 Bytes)
//
//=============================================================================
